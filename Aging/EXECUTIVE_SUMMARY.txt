  cfg.dataDir = '/data/path';
  state = Main_Aging(cfg);
  → Result: Clean output, essential plots saved

FULL DIAGNOSTICS (Debug mode):
  cfg = agingConfig();
  cfg.debug.level = "full";
  cfg.debug.plots = "all";
  cfg.debug.plotVisible = "on";
  state = Main_Aging(cfg);
  → Result: Verbose, all plots visible

  cfg = agingConfig();
  cfg.debug.level = "quiet";
  cfg.debug.plots = "none";
  for i = 1:100
    cfg.dataDir = getData(i);
    state = Main_Aging(cfg);
  end
  → Result: No spam, all results saved

   A: cfg.debug.plotVisible = "on";
